M√©todo de Simpson

Implementaci√≥n del m√©todo de Simpson para el c√°lculo aproximado de integrales definidas.

import numpy as np

def simpson_1_3(f, a, b, n):
    """
    Regla de Simpson 1/3 para integraci√≥n num√©rica
    f: funci√≥n a integrar
    a, b: l√≠mites de integraci√≥n
    n: n√∫mero de subintervalos (debe ser par)
    """
    if n % 2 != 0:
        raise ValueError("n debe ser par")
    
    h = (b - a) / n
    x = np.linspace(a, b, n+1)
    y = f(x)
    
    # Aplicar regla de Simpson
    resultado = y[0] + y[-1]
    resultado += 4 * np.sum(y[1:-1:2])
    resultado += 2 * np.sum(y[2:-1:2])
    resultado *= h / 3
    
    return resultado

# Ejemplo: integrar x^2 de 0 a 1
f = lambda x: x**2
integral = simpson_1_3(f, 0, 1, 100)
print(f"Resultado: {integral}")
print(f"Valor exacto: {1/3}")

Runge-Kutta

Soluci√≥n num√©rica de ecuaciones diferenciales ordinarias usando el m√©todo de Runge-Kutta de 4to orden.

import numpy as np
import matplotlib.pyplot as plt

def runge_kutta_4(f, y0, t0, tf, h):
    """
    M√©todo de Runge-Kutta de 4to orden
    f: dy/dt = f(t, y)
    y0: condici√≥n inicial
    t0, tf: tiempo inicial y final
    h: tama√±o de paso
    """
    n = int((tf - t0) / h)
    t = np.linspace(t0, tf, n+1)
    y = np.zeros(n+1)
    y[0] = y0
    
    for i in range(n):
        k1 = h * f(t[i], y[i])
        k2 = h * f(t[i] + h/2, y[i] + k1/2)
        k3 = h * f(t[i] + h/2, y[i] + k2/2)
        k4 = h * f(t[i] + h, y[i] + k3)
        
        y[i+1] = y[i] + (k1 + 2*k2 + 2*k3 + k4) / 6
    
    return t, y

# Ejemplo: dy/dt = -2y, y(0) = 1
f = lambda t, y: -2*y
t, y = runge_kutta_4(f, 1.0, 0, 2, 0.1)

plt.plot(t, y, 'b-', label='Num√©rico')
plt.plot(t, np.exp(-2*t), 'r--', label='Exacto')
plt.legend()
plt.show()

Trabajo 3: Ajuste de Curvas

M√≠nimos Cuadrados

Implementaci√≥n del m√©todo de m√≠nimos cuadrados para el ajuste de datos experimentales.

import numpy as np
import matplotlib.pyplot as plt

def minimos_cuadrados(x, y, grado):
    """
    Ajuste polinomial por m√≠nimos cuadrados
    x, y: datos experimentales
    grado: grado del polinomio
    """
    # Crear matriz de Vandermonde
    A = np.vander(x, grado + 1)
    
    # Resolver sistema normal
    coef = np.linalg.lstsq(A, y, rcond=None)[0]
    
    return coef

# Ejemplo
x = np.array([0, 1, 2, 3, 4, 5])
y = np.array([1.0, 2.1, 3.9, 6.2, 8.8, 12.1])

coef = minimos_cuadrados(x, y, 2)
print("Coeficientes:", coef)

# Graficar
x_fit = np.linspace(0, 5, 100)
y_fit = np.polyval(coef, x_fit)

plt.scatter(x, y, color='red', label='Datos')
plt.plot(x_fit, y_fit, 'b-', label='Ajuste')
plt.legend()
plt.show()
