import re

def analizar_funcion(funcion):
    funcion = funcion.replace(" ", "")
    variables = sorted(set(re.findall(r'[a-zA-Z]', funcion)))
    terminos = re.findall(r'[+-]?\d*[a-zA-Z]?\^?\d*', funcion)
    terminos = [t for t in terminos if t]
    coeficientes = []
    signos = []
    for t in terminos:
        if t.startswith('-'):
            signos.append('-')
        else:
            signos.append('+')
        c = re.findall(r'[+-]?\d+', t)
        if c:
            coef = int(c[0])
        else:
            coef = -1 if t.startswith('-') else 1
        coeficientes.append(coef)
    print(f"\nüîπ Funci√≥n ingresada: {funcion}")
    print(f"üî∏ Variables encontradas: {variables}")
    print(f"üî∏ T√©rminos: {terminos}")
    print(f"üî∏ Coeficientes: {coeficientes}")
    print(f"üî∏ Signos: {signos}")

print("=== ANALIZADOR DE FUNCIONES ALGEBRAICAS ===")

Trabajo 2: Interpolaci√≥n Polinomial

Lagrange

Implementaci√≥n del m√©todo de interpolaci√≥n polinomial de Lagrange para aproximar valores intermedios.

import numpy as np
import matplotlib.pyplot as plt

def interpolacion_lagrange(x, y, xi):
    """
    Interpolaci√≥n polinomial de Lagrange
    x: puntos conocidos (x)
    y: valores conocidos (y)
    xi: punto a interpolar
    """
    n = len(x)
    resultado = 0
    
    for i in range(n):
        termino = y[i]
        for j in range(n):
            if i != j:
                termino *= (xi - x[j]) / (x[i] - x[j])
        resultado += termino
    
    return resultado

# Ejemplo
x = np.array([0, 1, 2, 3])
y = np.array([1, 3, 2, 5])
xi = 1.5

yi = interpolacion_lagrange(x, y, xi)
print(f"Valor interpolado en x={xi}: {yi}")

import numpy as np

def eliminacion_gaussiana(A, b):
    """
    Resuelve Ax = b usando eliminaci√≥n gaussiana
    """
    n = len(b)
    Ab = np.column_stack([A, b]).astype(float)
    
    # Eliminaci√≥n hacia adelante
    for i in range(n):
        # Pivoteo parcial
        max_row = i + np.argmax(np.abs(Ab[i:, i]))
        Ab[[i, max_row]] = Ab[[max_row, i]]
        
        for j in range(i+1, n):
            factor = Ab[j, i] / Ab[i, i]
            Ab[j, i:] -= factor * Ab[i, i:]
    
    # Sustituci√≥n hacia atr√°s
    x = np.zeros(n)
    for i in range(n-1, -1, -1):
        x[i] = (Ab[i, -1] - np.dot(Ab[i, i+1:n], x[i+1:])) / Ab[i, i]
    
    return x

# Ejemplo
A = np.array([[2, 1, -1], [-3, -1, 2], [-2, 1, 2]])
b = np.array([8, -11, -3])
solucion = eliminacion_gaussiana(A, b)
print("Soluci√≥n:", solucion)
